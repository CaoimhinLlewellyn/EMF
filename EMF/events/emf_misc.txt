
# The home of truly random, miscellaneous events

namespace = emf_misc


# emf_misc.0 [Isis]
#
# Remove invalid claims on the Papacy.
character_event = {
	id = emf_misc.0
	desc = AI_EVENT
	hide_window = yes
	
	only_independent = yes
	
	trigger = {
		has_landed_title = e_hip
		or = {
			k_papal_state = {
				any_claimant = {
					is_theocracy = no
				}
			}
			d_fraticelli = {
				any_claimant = {
					is_theocracy = no
				}
			}
		}
	}
	
	immediate = {
		k_papal_state = {
			any_claimant = {
				limit = { is_theocracy = no }
				remove_claim = PREV
			}
		}
		d_fraticelli = {
			any_claimant = {
				limit = { is_theocracy = no }
				remove_claim = PREV
			}
		}
	}
}

# emf_misc.1 [Antipope]
#
# Remove claim on Papacy. CB supplement due to vanilla 2.3.2 bug.
character_event = {
	id = emf_misc.1
	desc = AI_EVENT
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		remove_claim = k_papal_state
		remove_claim = d_fraticelli
	}
}
