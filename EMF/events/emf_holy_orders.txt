# The Grand Master requests the right to build a castle in your demesne
character_event = {
	id = emf_holy.1250
	desc = OK
	picture = GFX_evt_moors
	border = GFX_event_normal_frame_religion
	
	hide_window = yes
	
	only_playable = yes
	
	trigger = {
		OR = {
			religion = sunni
			religion = zikri
			religion = yazidi
		}
		
		d_arsiyah = {
			holder_scope = {
				wealth = 300
				NOT = {
					realm_size = 10
				}
				NOT = {
					num_of_count_titles = 1
				}
			}
		}
		
		any_demesne_province = {
			has_empty_holding = yes
			port = yes
		}
		
		has_dlc = "Sons of Abraham"
		
		NOT = { has_character_modifier = expelled_d_arsiyah }
		NOT = {
			any_liege = {
				has_character_modifier = expelled_d_arsiyah
			}
		}
	}
	
	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.5
			d_arsiyah = {
				holder_scope = {
					NOT = { demesne_size = 1 }
				}
			}
		}
		modifier = {
			factor = 0.5
			d_arsiyah = {
				holder_scope = {
					NOT = { demesne_size = 2 }
				}
			}
		}
		modifier = {
			factor = 2.0
			d_arsiyah = {
				holder_scope = {
					realm_size = 5
				}
			}
		}
	}
	
	immediate = {
		random_demesne_province = {
			limit = {
				has_empty_holding = yes
				port = yes
			}
			province_event = {
				id = emf_holy.1251
			}
		}
	}
	
	option = {
		name = OK
	}
}
