# Isis bounce event
character_event = {
	id = emf_crier.1
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		FROM = { is_alive = no }
	}
	
	immediate = {
		log = "Isis got event"
	}
	
	option = { # FROM = Pope
		trigger = {
			k_papal_state = {
				any_previous_holder = { character = FROM }
			}
		}
		name = OK
		
		if = { # Pope was sacrificed
			limit = {
				FROM = { death_reason = death_sacrificed }
			}
			# Send event to dead pope? Or to a Christian ruler?
			narrative_event = { id = emf_crier.50 days = 7 random = 7 }
			log = "Isis sent sacrificed narrative event"
		}
	}
}
