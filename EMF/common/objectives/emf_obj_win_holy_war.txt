obj_win_holy_war = {
	type = character

	allow = {
		ai = no
		prisoner = no
		is_adult = yes
		is_ruler = yes
		not = { trait = incapable }
		war = no
		not = { has_character_modifier = emf_holy_victory_timer }
	}
	chance = {
		factor = 0
	}
	success = {
		has_character_modifier = emf_holy_victory_timer
	}
	abort = {
		or = {
			is_ruler = no
			prisoner = yes
			trait = incapable
		}
	}
	abort_effect = {
	}
	effect = {
		if = {
			limit = { not = { has_character_flag = obj_win_holy_war } }
			change_martial = 1
			piety = 40
		}
		if = {
			limit = { has_character_flag = obj_win_holy_war }
			piety = 20
		}
		set_character_flag = obj_win_holy_war
	}
}
