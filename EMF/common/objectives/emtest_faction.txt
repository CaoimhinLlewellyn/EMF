# Faction tags need to start with 'faction_'

# Supported types:
# liege_titles, liege_titles_w_claimant, character

# Generic Revolt Testing Faction
#
# Membership and leadership are completely scripted.
faction_emtest = {
	type = liege_titles
	
	warning_level = 0.7 # At this level of faction power we give player an alert. Set to 0 for no alerts.
	
	# Plotter scope
	potential = {
		always = no
		is_playable = yes
		independent = no
		primary_title = { higher_tier_than = baron }
		
		liege = {
			higher_tier_than = count
			not = {
				any_demesne_title = {
					temporary = yes
				}
			}
		}
	}
	
	# Target scope
	allow = {
		is_primary_holder_title = yes
		FROM = { has_character_flag = emtest_faction }
	}
	
	# Faction member scope (ROOT = joiner, FROM = target)
	allow_join = {
		ROOT = {
			is_playable = yes
			independent = no
		}
	}
	
	# AI creation weight
	chance = {
		factor = 500
		
		modifier = {
			factor = 0
			FROM = { not = { has_character_flag = emtest_faction_leader } }
		}
	}
	
	# AI membership weight: ROOT is the prospective member. FROM is the faction leader. FROMFROM is the target title or character.
	membership = {
		factor = 500
		
		modifier = {
			factor = 0
			FROM = { not = { has_character_flag = emtest_faction } }
		}
	}
	
	success = {
		always = no
	}
	
	abort = {
		always = no
	}
	
	abort_effect = {
	}
	
	effect = {
	}
}
