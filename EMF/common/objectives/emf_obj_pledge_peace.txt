obj_pledge_peace = {
	type = character

	allow = {
		ai = no
		higher_tier_than = baron
		prisoner = no
		not = { trait = incapable }
		war = no
		not = { has_character_modifier = emf_peace_pledge }
	}
	chance = {
		factor = 0
	}
	creation_effect = {
		add_character_modifier = { name = emf_peace_pledge duration = 5475 }
	}
	success = {
		custom_tooltip = {
			text = emf_ctt_obj_pledge_peace
			hidden_tooltip = {
				not = { has_character_modifier = emf_peace_pledge }
			}
		}
	}
	abort = {
		or = {
			is_primary_war_attacker = yes
			in_revolt = yes
		}
	}
	abort_effect = {
		if = {
			limit = { higher_tier_than = duke }
			prestige = -300
			piety = -150
		}
		if = {
			limit = { tier = duke }
			prestige = -200
			piety = -100
		}
		if = {
			limit = { tier = count }
			prestige = -100
			piety = -50
		}
		remove_character_modifier = emf_peace_pledge
		add_character_modifier = { name = emf_peace_pledge_aborted duration = 1825 }
	}
	effect = {
		if = {
			limit = { not = { has_character_flag = obj_pledge_peace } }
			change_stewardship = 2
		}
		if = {
			limit = { higher_tier_than = duke }
			prestige = 300
			piety = 150
		}
		if = {
			limit = { tier = duke }
			prestige = 200
			piety = 100
		}
		if = {
			limit = { tier = count }
			prestige = 100
			piety = 50
		}
		set_character_flag = obj_pledge_peace
	}
}
