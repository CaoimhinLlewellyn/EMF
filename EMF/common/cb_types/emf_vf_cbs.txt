
# [Standard type of] faction war to legalize the VF
vf_legalize_cooperation = {
	name = CB_NAME_LEGALIZE_COOPERATION
	war_name = WAR_NAME_LEGALIZE_COOPERATION
	sprite = 6
	truce_days = 1825
	attacker_can_call_allies = no
	can_ask_to_join_war = no
	major_revolt = yes
	
	can_use = {
		ROOT = {
			vassal_of = FROM
		}
	}
	
	is_valid = {
		FROM = { higher_tier_than = count }
		ROOT = {
			or = {
				liege = {
					character = PREV # either independent
				}
				liege = { 
					FROM = { 
						is_liege_or_above = PREV # or have shared liege
					}
				}
			}
		}
	}
	
	is_valid_title = {
		not = {
			has_law = vf_cooperation_1
			has_law = vf_cooperation_2
			has_law = vf_cooperation_3
		}
	}
	
	on_success = {
		FROM = {
			hidden_tooltip = { disband_event_forces = faction_loyalists }
		}
		any_attacker = {
			limit = { character = ROOT }
			participation_scaled_prestige = 100
			hidden_tooltip = {
				disband_event_forces = faction_revolters
			}
		}
		any_attacker = {
			limit = { NOT = { character = ROOT } }
			hidden_tooltip = {
				participation_scaled_prestige = 100
				disband_event_forces = faction_revolters
			}
		}
	}

	on_success_title = {
		add_law_w_cooldown = vf_cooperation_1
		
		# TODO: Determine if it is even necessary to copy laws to other
		# primary-potential titles to disallow primary-switching as a
		# law loophole. [While it's unclear whether hidden laws will
		# automatically get copied on primary switch, it seems regular
		# demesne laws like this would.]
		
		hidden_tooltip = {
			FROM = {
				primary_title = {
					FROM = {
						any_demesne_title = {
							limit = {
								not = {
									lower_tier_than = PREVPREV
									has_law = vf_cooperation_1
									has_law = vf_cooperation_2
									has_law = vf_cooperation_3
								}
							}
							add_law_w_cooldown = vf_cooperation_1
						}
					}
				}
			}
		}
		
		# TODO:
		# Schedule VF faction leadership election
	}

	on_fail = {
		ROOT = {
			prestige = -100
		}
		any_defender = {
			limit = { character = FROM }
			participation_scaled_prestige = 50
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { participation_scaled_prestige = 50 }
		}
		
		hidden_tooltip = {
			any_attacker = {
				disband_event_forces = faction_revolters
			}
		}
		FROM = { hidden_tooltip = { disband_event_forces = faction_loyalists } }
	}

	on_reverse_demand = {
		ROOT = {
			prestige = -200
			prisoner = FROM
		}
		any_defender = {
			limit = { character = FROM }
			participation_scaled_prestige = 100
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { participation_scaled_prestige = 100 }
		}
		
		hidden_tooltip = {
			any_attacker = {
				limit = { liege = { character = FROM } }
				disband_event_forces = faction_revolters
			}
		}
		
		FROM = { hidden_tooltip = { disband_event_forces = faction_loyalists } }
	}
	
	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	attacker_ai_defeat_worth = {
		factor = 100
	}
	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	defender_ai_defeat_worth = {
		factor = 100
	}
}
